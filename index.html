  <title>TVS Diode Voltage Range Comparison</title>

  <script src="https://cdnjs.cloudflare.com/ajax/libs/prop-types/15.8.1/prop-types.min.js"></script>

  <script src="https://cdnjs.cloudflare.com/ajax/libs/react/18.2.0/umd/react.production.min.js"></script>

  <script src="https://cdnjs.cloudflare.com/ajax/libs/react-dom/18.2.0/umd/react-dom.production.min.js"></script>

  <script src="https://cdnjs.cloudflare.com/ajax/libs/babel-standalone/7.23.2/babel.min.js"></script>

  <script src="https://unpkg.com/papaparse@latest/papaparse.min.js"></script>

  <script src="https://cdnjs.cloudflare.com/ajax/libs/chrono-node/1.3.11/chrono.min.js"></script>

  <script src="https://cdnjs.cloudflare.com/ajax/libs/recharts/2.15.0/Recharts.min.js"></script>

  <script src="https://cdn.tailwindcss.com"></script>

</head>

<body class="bg-gray-100 font-sans">

  <div id="root" class="container mx-auto p-4"></div>

 

  <script type="text/babel">

    // Initializing React and Recharts components

    const { useState, useEffect } = React;

    const { BarChart, Bar, XAxis, YAxis, Tooltip, ResponsiveContainer, CartesianGrid, Legend } = Recharts;

 

    // Defining the main App component

    function App() {

      const [data, setData] = useState(null);

      const [largestChange, setLargestChange] = useState(null);

 

      // Processing and cleaning CSV data

      const processAndCleanData = (rawData) => {

        return rawData

          .filter(row => row["Series Name"] && row["Existing voltage range"] && row["New voltage range"])

          .map(row => {

            const existingRange = row["Existing voltage range"].replace(" V", "").split(" to ");

            const newRange = row["New voltage range"].replace(" V", "").split(" to ");

            const minExisting = parseFloat(existingRange[0]);

            const maxExisting = parseFloat(existingRange[1]);

            const minNew = parseFloat(newRange[0]);

            const maxNew = parseFloat(newRange[1]);

            if (isNaN(minExisting) || isNaN(maxExisting) || isNaN(minNew) || isNaN(maxNew)) {

              return null;

            }

            return {

              series: row["Series Name"],

              minExisting,

              maxExisting,

              existingRange: maxExisting - minExisting,

              minNew,

              maxNew,

              newRange: maxNew - minNew,

              rangeChange: Math.abs((maxExisting - minExisting) - (maxNew - minNew))

            };

          })

          .filter(item => item !== null);

      };

 

      // Loading and parsing CSV data

      useEffect(() => {

        const csv = loadFileData("TVS CSV.csv");

        Papa.parse(csv, {

          header: true,

          skipEmptyLines: true,

          transformHeader: header => header.trim().replace(/^"|"$/g, ''),

          transform: (value, header) => value.trim().replace(/^"|"$/g, ''),

          complete: (results) => {

            const cleanedData = processAndCleanData(results.data);

            setData(cleanedData);

            // Finding the series with the largest change in voltage range

            const maxChangeSeries = cleanedData.reduce((max, item) =>

              item.rangeChange > (max.rangeChange || 0) ? item : max, {});

            setLargestChange(maxChangeSeries);

          },

          error: (err) => console.error('Error parsing CSV:', err)

        });

      }, []);

 

      // Rendering loading state

      if (!data) {

        return (

          <div className="text-center text-lg text-gray-600 mt-10">

            Loading data...

          </div>

        );

      }

 

      // Rendering the report

      return (

        <div className="bg-white shadow-lg rounded-lg p-6">

          {/* Summary section */}

          <h1 className="text-3xl font-bold text-blue-800 mb-4">TVS Diode Voltage Range Comparison</h1>

          <p className="text-gray-700 mb-4">

            This report compares the existing and new reverse standoff voltage ranges for various TVS diode series.

            The chart below displays both ranges side by side for each series, highlighting changes in voltage coverage.

            <br />

            <strong>Interesting Fact:</strong> The {largestChange.series} series shows the largest change in voltage range

            ({largestChange.rangeChange} V), indicating a significant adjustment in its application scope.

          </p>

 

          {/* Visualization section */}

          <h2 className="text-2xl font-semibold text-blue-700 mb-3">Voltage Range Comparison by Series</h2>

          <div className="w-full h-[400px]">

            <ResponsiveContainer width="100%" height="100%">

              <BarChart data={data} layout="vertical" margin={{ top: 10, right: 30, left: 50, bottom: 10 }}>

                <CartesianGrid strokeDasharray="3 3" />

                <XAxis type="number" domain={[0, 600]} label={{ value: 'Voltage (V)', position: 'bottom', offset: 0 }} />

                <YAxis dataKey="series" type="category" width={100} />

                <Tooltip formatter={(value, name) => `${value} V`} />

                <Legend />

                <Bar dataKey="existingRange" name="Existing Range" fill="#3B82F6" />

                <Bar dataKey="newRange" name="New Range" fill="#10B981" />

              </BarChart>

            </ResponsiveContainer>

          </div>

 

          {/* Conclusion section */}

          <h2 className="text-2xl font-semibold text-blue-700 mt-6 mb-3">Conclusion</h2>

          <p className="text-gray-700">

            The visualization highlights the differences between existing and new voltage ranges across TVS diode series.

            The {largestChange.series} series, with a range change of {largestChange.rangeChange} V,

            indicates a significant shift in its voltage suppression capabilities. Series like SMAJ & P4SMA and SMCJE

            show no change, maintaining their original ranges, while others like 15BJ have notably reduced ranges,

            suggesting more specialized applications.

          </p>

        </div>

      );

    }

 

    // Rendering the app

    const root = ReactDOM.createRoot(document.getElementById('root'));

    root.render(<App />);

  </script>

</body>
